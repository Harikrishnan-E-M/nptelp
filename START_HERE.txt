✨ HOW IT WORKS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

USER JOURNEY:

1. User visits http://localhost:3000
   ↓
2. Sees list of academic years (YearsList.jsx)
   ↓
3. Clicks "Add Academic Year" or selects existing year
   ↓
4. For existing year: Clicks "View Statistics"
   ↓
5. Sees Statistics page (Statistics.jsx)
   ↓
6. Can upload CSV file
   ↓
7. Backend (server.js) parses CSV
   ↓
8. Data stored in Sanity via sanityClient
   ↓
9. Statistics calculated and displayed
   ↓
10. User can filter and search data

🔌 API ENDPOINTS (All in backend/server.js)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

GET /api/years
  Returns all academic years

POST /api/years
  Create new academic year
  Body: { yearLabel, startYear, endYear, description }

GET /api/statistics/:yearId
  Get statistics for specific year

POST /api/upload/:yearId
  Upload CSV file for year
  Body: FormData with file

DELETE /api/years/:yearId
  Delete year and all its data

GET /health
  Check server status


💾 DATA STORED IN SANITY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

academicYear (Document Type)
  • yearLabel: "2023-24"
  • startYear: 2023
  • endYear: 2024
  • description: (optional)
  • dataCount: auto-updated

nptelData (Document Type)
  • year: reference to academicYear
  • batch, regNo, name
  • semester, courseCode, courseTitle
  • credit, score
  • examMonth, examYear, certId
  • status: Accepted/Rejected/Pending
